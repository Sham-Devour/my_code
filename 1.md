#include <cstdio>
#include <iostream>
#include <cstring>
#define int long long
using namespace std;

int numa[25], numb[25], dpa[25][25], dpb[25][25];
inline int dfsa(int pos, int cnt, int limit, int gb) {
	if (!pos) return cnt;
	int &ans = dpa[pos][cnt];
	if (!limit && ~ans) return ans;
	int up = limit ? numa[pos] : 9, res = 0;
	for (int i = 0; i <= up; i++)
		res += dfsa(pos - 1, cnt + (i == gb), limit && i == up, gb);
	return limit ? res : ans = res;
}
inline int dfsb(int pos, int cnt, int limit, int gb) {
	if (!pos) return cnt;
	int &ans = dpb[pos][cnt];
	if (!limit && ~ans) return ans;
	int up = limit ? numb[pos] : 9, res = 0;
	for (int i = 0; i <= up; i++)
		res += dfsb(pos - 1, cnt + (i == gb), limit && i == up, gb);
	return limit ? res : ans = res;
}
inline int dfs0a(int pos, int cnt, int limit, int gb, int dgb) {
	if (!pos) return cnt;
	int &ans = dpa[pos][cnt];
	if (!limit && !dgb && ~ans) return ans;
	int up = limit ? numa[pos] : 9, res = 0;
	for (int i = 0; i <= up; i++)
		res += dfs0a(pos - 1, cnt + (i == gb && !dgb), limit && i == up, gb, dgb && (!i));
	return limit || dgb ? res : ans = res;
}
inline int dfs0b(int pos, int cnt, int limit, int gb, int dgb) {
	if (!pos) return cnt;
	int &ans = dpb[pos][cnt];
	if (!limit && !dgb && ~ans) return ans;
	int up = limit ? numb[pos] : 9, res = 0;
	for (int i = 0; i <= up; i++)
		res += dfs0b(pos - 1, cnt + (i == gb && !dgb), limit && i == up, gb, dgb && (!i));
	return limit || dgb ? res : ans = res;
}

signed main() {
	int a, b;
	scanf("%lld %lld", &a, &b);
	a--;
	while (a) numa[++numa[0]] = a % 10, a /= 10;
	while (b) numb[++numb[0]] = b % 10, b /= 10;
	memset(dpa, -1, sizeof(dpa));
	memset(dpb, -1, sizeof(dpb));
	printf("%lld\n", dfs0b(numb[0], 0, 1, 0, 1) - dfs0a(numa[0], 0, 1, 0, 1));
	for (int i = 1; i <= 9; i++) {
		memset(dpa, -1, sizeof(dpa));
		memset(dpb, -1, sizeof(dpb));
		printf("%lld\n", dfsb(numb[0], 0, 1, i) - dfsa(numa[0], 0, 1, i));
	}
	return 0;
}
